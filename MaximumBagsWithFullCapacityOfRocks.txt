//approach - 2
//time complexity - O(nlogn)
//space complexity - O(logn)
class Solution {
public:
    int maximumBags(vector<int>& capacity, vector<int>& rocks, int additionalRocks) {
        int n = capacity.size();

        // Calculate the "empty space" in each bag and store it back in 'capacity'
        // This saves space by reusing the input vector instead of creating a new one
        for(int i = 0; i < n; i++) {
            capacity[i] = capacity[i] - rocks[i];
        }

        // Greedy Strategy: Sort bags by the amount of rocks needed to fill them
        // Filling the bags that need the fewest rocks first maximizes the count of full bags
        sort(capacity.begin(), capacity.end());

        int fullBags = 0;
        for(int i = 0; i < n; i++) {
            // Check if we have enough additional rocks to completely fill the current bag
            if(capacity[i] <= additionalRocks) {
                fullBags++;
                additionalRocks -= capacity[i];
            } else {
                // Since the list is sorted, if we can't fill this bag, we can't fill any subsequent bags
                break;
            }
        }

        return fullBags;
    }
};
