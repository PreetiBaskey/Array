//approach - 1
//time complexity - O(nlogn)
//space complexity - O(1)
class Solution {
public:
    int findMinArrowShots(vector<vector<int>>& points) {
        int n = points.size();
        sort(begin(points), end(points)); //sort based on the points[i][0]

        vector<int> prev = points[0]; //consider points[0] as first prev
        int count = 1; //count it 1 for first prev to burst

        for(int i = 1; i < n; i++) { 
            int prevStartPoint = prev[0];
            int prevEndPoint = prev[1];

            int currStartPoint = points[i][0];
            int currEndPoint = points[i][1];

            if(currStartPoint > prevEndPoint) {
                count++;
                prev = points[i];
            }
            else { //calculate overlap points
                prev[0] = max(prevStartPoint, currStartPoint);
                prev[1] = min(prevEndPoint, currEndPoint);
            }
        }

        return count;
    }
};
