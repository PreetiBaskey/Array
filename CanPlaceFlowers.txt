//approach - 1
//time complexity - O(m)
//space complexity - O(1)
class Solution {
public:
    bool canPlaceFlowers(vector<int>& flowerbed, int n) {
        int m = flowerbed.size();

        // If no flowers need to be planted, it's always possible
        if(n == 0) {
            return true;
        }

        for(int i = 0; i < m; i++) {
            // Check if current plot is empty
            if(flowerbed[i] == 0) {
                // Check if left and right plots are empty or out of bounds
                bool leftSide = (i == 0) || (flowerbed[i - 1] == 0);
                bool rightSide = (i == m - 1) || (flowerbed[i + 1] == 0);

                // If both sides are clear, plant a flower here
                if(leftSide && rightSide) {
                    flowerbed[i] = 1;
                    n--;

                    // Return immediately if all flowers are planted
                    if(n <= 0) {
                        return true;
                    }
                }
            }
        }

        // Return whether we managed to plant all required flowers
        return n <= 0;
    }
};
