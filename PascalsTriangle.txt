//approach - 1
//time complexity - O(n^2)
//space complexity - O(n^2)
class Solution {
public:
    vector<vector<int>> generate(int numRows) {
        vector<vector<int>> result;

        for(int i = 0; i < numRows; i++) {
            // Create a row with 'i + 1' elements, all initialized to 1
            // This automatically handles the first and last '1' of every row
            vector<int> row(i + 1, 1);
            
            // Fill the middle elements (from index 1 to i-1)
            // Each element is the sum of the two elements directly above it in the previous row
            for(int j = 1; j < i; j++) {
                row[j] = result[i - 1][j] + result[i - 1][j - 1];
            }

            // Add the completed row to our 2D result vector
            result.push_back(row);
        }

        return result;
    }
};
