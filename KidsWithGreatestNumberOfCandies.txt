//approach - 1
//time complexity - O(n)
//space complexity - O(1)
class Solution {
public:
    vector<bool> kidsWithCandies(vector<int>& candies, int extraCandies) {
        int n = candies.size();
        
        // Initialize the result vector with 'false'
        vector<bool> result(n, false); 

        // Step 1: Find the current maximum candies any child has.
        // *max_element is an O(n) operation.
        int maxCandy = *max_element(candies.begin(), candies.end());

        // Step 2: Iterate through each child to check the condition.
        for(int i = 0; i < n; i++) {
            // If the child's current candies + extra candies is 
            // greater than or equal to the current maximum, they meet the requirement.
            if(candies[i] + extraCandies >= maxCandy) {
                result[i] = true;
            }
        }

        return result;
    }
};
